class Node:
    def __init__(self, e, n):
        self.element = e
        self.next = n





class LinkedList:
  def __init__(self, a):
   
        self.head = None
       
        if type(a) == LinkedList:
            self.head = a.head
            
        if type(a) == list:
            i = 0

  # # Count the number of nodes in the list
  def countNode(self):
        # To Do
        count = 0
        n = self.head
        while n is not None:
            count += 1
            n = n.next
        return count

  # returns the reference of the Node at the given index. For invalid index return None.
  def nodeAt(self, idx):
    n = self.head
    count = 0
    while n is not None:
      if count == idx:
        return n
        # print(n.element)
      count += 1
      n= n.next

    n = self.head
    count = 0
    while n is not None:
      if count == idx:
        return n.element
      
        
      count += 1
      n= n.next

    n2 = self.get(idx)  
    n = self.head

    while n is not None:
      if count == idx:

  def indexOf(self, elem):
    n = self.head
    count = 0
    while n is not None:
      if n.element == elem:
        p = count
      else:
        p = -1
      count += 1
      n= n.next
    return p
    
    

  def contains(self, elem):
    n = self.head
    count = 0
    while n is not None:
      if n.element == elem:
        p = True
      else:
        p = False
      count += 1
      n= n.next
    return p


  def copyList(self):
    n = self.head
    list1 = []
    while n is not None:
          list1.append(n.element)
          n=n.next
    
    return list1
          
  
  def reverseList(self):
    copy1 = LinkedList(None)
    count1 = 0
    count2 = self.countNode() - 1
    copy1.head = None

    while(count2 >= 0):
        n = self.nodeAt(count2)
        copy1.insert(n.element, count1)
        count2 -= 1
        count1 += 1
    return copy1

 
  def insert(self, elem, idx):

        count = 0
        nnode = Node(elem, None)
        if idx == 0 :
            nnode.next = self.head
            self.head = nnode
            return
        
        n = self.head
        while n is not None:
            if count == idx-1 :
                nnode.next = n.next
                n.next = nnode
                return
            count += 1
            n = n.next
         
   

  def remove(self, idx):
        count = 0
        n = self.head
        if idx == 0:
            ele1 = self.head.element
            self.head = self.head.next
            return ele1

        
        while n is not None:
            if count == idx-1:
                ele1 = n.next.element
                n.next = n.next.next
                return ele1
            count += 1
            n = n.next
  
 
  def rotateLeft(self):
        count = 0
        n = self.head
        re_head = self.remove(0)
        while n is not None:
            count += 1
            n = n.next
        
        self.insert(re_head, count)


  
  
 
  def rotateRight(self):
        point_last = self.remove(self.countNode()-1)
        self.insert(point_last, 0)
